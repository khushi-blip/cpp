Q: WRITE UDF TO PRINT LENTGH OF STRING
#include<stdio.h>
int xstrlen(char*);
int main()
{
  char arr[]="hello";
  char bye[]="bye";
  int len1,len2;
  len1= xstrlen(arr);
  len2= xstrlen(bye);
  printf("string =%s length =%d\n",arr,len1);
  printf("string =%s length =%d\n",bye,len2);
  return 0;
}
int xstrlen(char *s)
{
  int length =0;
  while(*s!='\0'){
    length ++;
    s++;
  }
  return(length);
}

Q: FOR UPPER CASE

#include<stdio.h>

void strupr(char *s) {
    int i = 0;
    while (s[i] != '\0') { 
        if (s[i] >= 97 && s[i] <= 122) {  
            s[i] = s[i] - 32; 
        }
        i++;
    }
}
int main() {
    char ch[] = "hello";  
    printf("Original string is = %s\n", ch);
    
    strupr(ch);  
    
    printf("String is upper case = %s\n", ch);  
    
    return 0;

}
